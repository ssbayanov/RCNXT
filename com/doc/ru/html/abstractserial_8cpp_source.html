00442 void AbstractSerialPrivate::setupSerialNotifiers()
00443 {
00444     Q_Q(AbstractSerial);
00445     if (serialEngine) {
00446         switch ((q->openMode()) & (~QIODevice::Unbuffered)) {
00447             case QIODevice::ReadOnly:
00448                 serialEngine->setReadNotificationEnabled(true);
00449                 QObject::connect(serialEngine, SIGNAL(readNotification()), q, SLOT(canReadNotification()));
00450                 break;
00451             case QIODevice::WriteOnly:
00452                 break;
00453             case QIODevice::ReadWrite:
00454                 serialEngine->setReadNotificationEnabled(true);
00455                 QObject::connect(serialEngine, SIGNAL(readNotification()), q, SLOT(canReadNotification()));
00456                 break;
00457             default:
00458                 ;
00459         }
00460     }
00461 }

00471 {
00472     Q_Q(AbstractSerial);
00473     if (waitForReadyReadCalled) {
00474         emittedReadyRead = true;
00475         emit q->readyRead();
00476         waitForReadyReadCalled = false;
00477         return true;

00479     if (!emittedReadyRead) {
00480         emittedReadyRead = true;

01368    Returns the difference between msecs and elapsed. If msecs is -1,

01371 static int qt_timeout_value(int msecs, int elapsed)
01372 {
01373     if (msecs == -1)

01375     int timeout = msecs - elapsed;
01376     return (timeout < 0) ? 0 : timeout;

01393 bool AbstractSerial::waitForReadyRead(int msecs)
01394 {
01395     Q_D(AbstractSerial);
01396 
01397     if (!isOpen()) {
01398         emitStatusString(EDeviceIsNotOpen);
01399         return false;
01400     }
01401 
01402     QTime stopWatch;
01403     stopWatch.start();
01404 
01405     forever {
01406         bool readyToRead = false;
01407         bool readyToWrite = false;
01408         if (!d->serialEngine->waitForReadOrWrite(&readyToRead, &readyToWrite, true, false,
01409                                                     qt_timeout_value(msecs, stopWatch.elapsed()))) {
01410             emitStatusString(EWaitReadyReadTimeout);
01411             return false;
01412         }
01413         if (readyToRead) {
01414             if (bytesAvailable() > 0) {
01415                 d->waitForReadyReadCalled = true;
01416                 if (d->canReadNotification()) {
01417                     return true;
01418                 }
01419             }//bytesAvailable
01420         }//readyToRead
01421     }
01422 }
