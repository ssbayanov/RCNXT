00024 #if QT_VERSION >= 0x040300

00026 #else
00027     quint8 n=0;
00028     for (int i=0;i<data.size();i++) {

00030         if ((n >= 0) && (n <= 15)) baTmp.append(QByteArray::number(0, 16).toUpper());
00031         baTmp.append(QByteArray::number(n, 16).toUpper());
00032     }
00033 #endif
00034
